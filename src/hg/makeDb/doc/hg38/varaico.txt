# move bigbeds to track version directory:
mkdir /hive/users/jbirgmei/TRACK_VERSIONS/v5_20260527

# archive the previous release (release 2, "release 2 (20251110)") to the
# download archive before copying in the new files:
for db in hg19 hg38; do
    mkdir /hive/data/genomes/$db/bed/varaico.v2_20251110
    cp /hive/data/genomes/$db/bed/varaico/varaico.bb \
       /hive/data/genomes/$db/bed/varaico.v2_20251110/$db.varaico.v2_20251110.bb
    cp /hive/data/genomes/$db/bed/varaico/varaicoSuppl.bb \
       /hive/data/genomes/$db/bed/varaico.v2_20251110/$db.varaicoSuppl.v2_20251110.bb
    mkdir -p /usr/local/apache/htdocs-hgdownload/goldenPath/archive/$db/varaico/v2_20251110
    ln -s /hive/data/genomes/$db/bed/varaico.v2_20251110/$db.varaico.v2_20251110.bb \
          /usr/local/apache/htdocs-hgdownload/goldenPath/archive/$db/varaico/v2_20251110/$db.varaico.v2_20251110.bb
    ln -s /hive/data/genomes/$db/bed/varaico.v2_20251110/$db.varaicoSuppl.v2_20251110.bb \
          /usr/local/apache/htdocs-hgdownload/goldenPath/archive/$db/varaico/v2_20251110/$db.varaicoSuppl.v2_20251110.bb
done

# copy bigbeds:
cp /hive/users/jbirgmei/TRACK_VERSIONS/v5_20260527/hg19/normalized_variants.hg19.bb /hive/data/genomes/hg19/bed/varaico/varaico.bb

cp /hive/users/jbirgmei/TRACK_VERSIONS/v5_20260527/hg38/normalized_variants_supplements.hg38.bb /hive/data/genomes/hg38/bed/varaico/varaicoSuppl.bb
