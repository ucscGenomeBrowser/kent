#!/usr/bin/env perl

use strict;

    my $swapMeasure = "N/A";
    if ( -s $swapFile ) {
	$swapMeasure =

    chomp $orgName;
    if (length($orgName) < 1) {
        $orgName="N/A";

    ++$count;
    printf "# %02d  %.4f (%% %06.3f) (%% %06.3f) - %s %s\n", $count, $dist,
        $chainLinkMeasure, $swapMeasure, $orgName, $D;
}
close (FH);
